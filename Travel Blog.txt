<script>
// FILTER
function filterSelection(category) {
    let items = document.getElementsByClassName("blog-post");

    for (let i = 0; i < items.length; i++) {
        if (category === "all") {
            items[i].style.display = "block";
        } else {
            items[i].style.display = items[i].classList.contains(category) ? "block" : "none";
        }
    }
}

// AI ASSISTANT
function askAI() {
    let input = document.getElementById("user-input").value.toLowerCase();
    let output = document.getElementById("chat-output");

    let response = "I can help with travel tips!";

    if (input.includes("pune")) {
        response = "Visit Sinhagad Fort and Shaniwar Wada in Pune!";
    }
    else if (input.includes("goa")) {
        response = "Best time to visit Goa is Nov‚ÄìFeb.";
    }
    else if (input.includes("paris")) {
        response = "Paris is famous for Eiffel Tower and cafes.";
    }

    output.innerHTML += "<p><b>You:</b> " + input + "</p>";
    output.innerHTML += "<p><b>AI:</b> " + response + "</p>";

    document.getElementById("user-input").value = "";
}
</script>
